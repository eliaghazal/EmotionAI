Chapter 04 — Seeing Faces

Face Detection with OpenCV 


# core/predictor.py
def _preprocess_for_model(self, crop, model_type):
    if model_type == "cnn":
        # Grayscale + CLAHE contrast fix
        gray = cv2.cvtColor(crop, cv2.COLOR_BGR2GRAY)
        gray = self._clahe.apply(gray)  # webcam → FER2013 distribution
        resized = cv2.resize(gray, (48, 48))
        return (resized / 255.0)[..., np.newaxis]  # shape (48,48,1)
    else:
        # MobileNetV2: RGB, scale to [-1, 1]
        resized = cv2.resize(crop, (96, 96))
        rgb = cv2.cvtColor(resized, cv2.COLOR_BGR2RGB)
        return (rgb / 255.0) * 2.0 - 1.0  # shape (96,96,3)
CLAHE

Contrast Limited Adaptive Histogram Equalisation — aligns webcam contrast with FER2013's training distribution

Two preprocessors

CNN gets grayscale [0,1] · MobileNetV2 must get RGB [−1,1] (trained that way — wrong range → 11% accuracy)

# SE Attention — channel gating
se = layers.GlobalAveragePooling2D()(x)   # squeeze: 12×12×128 → 128
se = layers.Dense(16, activation="relu")(se)
se = layers.Dense(128, activation="sigmoid")(se)  # per-channel gate 0–1
se = layers.Reshape((1, 1, 128))(se)
x  = layers.Multiply()([x, se])   # excite: amplify important channels
    

# Residual connection
shortcut = layers.Conv2D(128, 1, padding="same")(x)
shortcut = layers.BatchNormalization()(shortcut)
# ... two 3×3 convolutions + SE ...
x = layers.Add()([x, shortcut])  # skip connection
x = layers.ReLU()(x)
Chapter 06 — Standing on Giants

MobileNetV2 Transfer Learning

Phase 1 — 20 epochs Train the head only MobileNetV2 base is frozen. Only the new Dense(256) → Dense(7) head is updated. LR = 1e-3 (aggressive, safe because base is locked).
↓
Phase 2 — 40 epochs Fine-tune the top layers Last 30 layers of MobileNetV2 are unfrozen. LR = 1e-5 (tiny — don't destroy ImageNet knowledge). + L2 regularisation + Dropout(0.5). 

# models/train_transfer.py
def unfreeze_top_layers(model, cfg):
    base = model.layers[1]  # MobileNetV2
    base.trainable = True
    # Keep the bottom layers frozen
    for layer in base.layers[:-cfg.unfreeze_layers]:
        layer.trainable = False

    model.compile(
        optimizer=Adam(lr=cfg.phase2_lr),  # 1e-5
        loss=CategoricalCrossentropy(
            label_smoothing=0.1),
    )

# Critical preprocessing
def apply_mobilenet_scale(X):
    # MobileNetV2 MUST receive [-1, 1]
    # Using [0,1] → stuck at ~11% accuracy
    return (X * 2.0 - 1.0).astype(np.float32)

Training Pipeline 


# utils/dataset_loader.py — tf.data with MixUp
def build_tf_dataset(X, y, batch_size=64,
                     augment=True, mixup_alpha=0.2):
    ds = tf.data.Dataset.from_tensor_slices((X, y))
    ds = ds.shuffle(10_000, reshuffle_each_iteration=True)

    if augment:
        def _aug(img, lbl):
            img = tf.image.random_flip_left_right(img)
            img = tf.image.random_brightness(img, 0.15)
            img = tf.image.random_contrast(img, 0.85, 1.15)
            # Zoom: crop then resize
            crop_h = tf.random.uniform((), 43, 49, tf.int32)
            img = tf.image.random_crop(img, [crop_h, crop_h, 1])
            img = tf.image.resize(img, [48, 48])
            return tf.clip_by_value(img, 0.0, 1.0), lbl
        ds = ds.map(_aug, num_parallel_calls=AUTOTUNE)

    ds = ds.batch(batch_size, drop_remainder=True)

    if mixup_alpha > 0:  # MixUp: blend two samples
        def _mixup(imgs, lbls):
            lam = tf.random.uniform(())
            lam = tf.maximum(lam, 1.0 - lam)  # lam ≥ 0.5
            idx = tf.random.shuffle(tf.range(tf.shape(imgs)[0]))
            return (lam*imgs + (1-lam)*tf.gather(imgs,idx),
                    lam*lbls + (1-lam)*tf.gather(lbls,idx))
        ds = ds.map(_mixup, num_parallel_calls=AUTOTUNE)

    return ds.repeat().prefetch(AUTOTUNE)  # never exhausts
MixUp

Blends pairs of images and labels — creates smooth decision boundaries, reduces overfitting

Label Smoothing

ε = 0.1 prevents overconfident predictions on FER2013's noisy labels

.repeat() Fix

Critical Keras 3.x bug: ImageDataGenerator exhausted every other epoch. tf.data + .repeat() solved it.

# core/predictor.py
def _infer(self, crop_bgr):
    face_input = self._preprocess_crop(crop_bgr)
    probs = self._model.predict(np.expand_dims(face_input, 0))[0]

    if self._ensemble_model is not None:
        ens_input = self._preprocess_for_model(crop_bgr, self._ensemble_model_type)
        ens_probs = self._ensemble_model.predict(np.expand_dims(ens_input, 0))[0]
        probs = (probs + ens_probs) / 2.0    # ← simple average

    # Temperature scaling: soften the distribution (T=4.0)
    # Without this: "99% Happy" — user sees nothing interesting
    # With this: "55% Happy, 20% Surprise, 15% Neutral" — full story
    logits = np.log(np.clip(probs, 1e-7, 1.0))
    scaled = logits / 4.0
    exp = np.exp(scaled - scaled.max())
    return (exp / exp.sum()).astype(np.float32), face_input

# core/gradcam.py — gradient tape to find important pixels
with tf.GradientTape() as tape:
    tape.watch(inp_tensor)
    conv_outputs, predictions = grad_model(inp_tensor)
    class_channel = predictions[:, predicted_class]   # e.g. class 3 = Happy

grads      = tape.gradient(class_channel, conv_outputs)
pooled     = tf.reduce_mean(grads, axis=(0, 1, 2))    # avg importance per channel
heatmap    = conv_outputs[0] @ pooled[..., tf.newaxis] # weighted feature map
heatmap    = np.maximum(heatmap, 0)                    # ReLU: keep positive
heatmap   /= heatmap.max()                             # normalise to [0,1]

The Bugs That Almost Beat Us

Challenges & Solutions

🐛 Keras 3.x epoch-boundary bug

Symptom: Every other epoch received only 1 step of data ("ran out of data")

✅ Fix

Replaced ImageDataGenerator.flow() with tf.data.Dataset.repeat()

🐛 MobileNetV2 at 11% accuracy

Symptom: Trained for hours, stuck near random chance regardless

✅ Fix

Model expects [-1, 1] range. Training used [0, 1]. Added x * 2 − 1 scaling.

🐛 Grad-CAM crash: "slice index out of bounds"

Symptom: predictions[:, 4] failed — "dimension 1 has size 3"

✅ Fix

Nested model's layer.get_layer().output is disconnected from outer graph. Used layer.output instead.

🐛 ReduceLROnPlateau crash at epoch 6

Symptom: TypeError: LearningRateSchedule is not settable

✅ Fix

CosineDecay makes LR non-settable in Keras 3. Removed ReduceLROnPlateau entirely — CosineDecay handles annealing.
